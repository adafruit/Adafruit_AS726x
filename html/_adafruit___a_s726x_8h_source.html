314  struct control_setup {
315 
316  uint8_t unused : 1;
317 
318  /* 1: data ready to be read, sets int active if int is enabled */
319  uint8_t DATA_RDY : 1;
320 
321  /* conversion type
322  * 0b00 = Mode 0
323  * 0b01 = Mode 1
324  * 0b10 = Mode 2
325  * 0b11 = Mode 3 One shot
326  */
327  uint8_t BANK : 2;
328 
329  /* Channel gain setting (all channels)
330  * 0b00 = 1x
331  * 0b01 = 3.7x
332  * 0b10 = 16x
333  * 0b11 = 64x
334  */
335  uint8_t GAIN : 2;
336 
337  /* enable or disable interrupt */
338  uint8_t INT : 1;
339  uint8_t RST : 1;
340 
341  uint8_t get() {
342  return ( (DATA_RDY << 1) | (BANK << 2) | (GAIN << 4) | (INT << 6) | (RST << 7) );
343  };
344  };
